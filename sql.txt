SELECT payment, SUM(sales) AS total_sales
FROM purchases
GROUP BY payment;

Exercise 1: 
SELECT category, SUM(sales)
FROM purchases
GROUP BY category;

Exercise 2: 
SELECT category, COUNT(*)
FROM purchases
GROUP BY category;

Exercise 3: 
SELECT category, COUNT(*)
FROM purchases
WHERE date IS NOT NULL AND time IS NOT NULL AND item IS NOT NULL AND category IS NOT NULL AND sales IS NOT NULL AND payment IS NOT NULL
GROUP BY category;

Exercise 4: 
SELECT category, SUM(sales)
FROM purchases
WHERE category IN ('Computers', 'Cameras', 'Video Games')
GROUP BY category;

Exercise 5: 
SELECT category, COUNT(*)
FROM purchases
GROUP BY category
HAVING COUNT(*) > 114;

Exercise 6: 
SELECT category, AVG(sales)
FROM purchases
GROUP BY category;
